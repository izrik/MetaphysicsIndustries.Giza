
expr = '\''? ( s-expr | number | string | identifier );

s-expr = '(' expr* ')';

<token, mind whitespace> number = (
    '0b' [01]+ | 
    '0o' [01234567]+ | 
    '0x' [\dabcdef]+ | 
    [\d]+ ( '.' [\d]+ )? ( [eE] [+-]? [\d]+ )? );

<token, mind whitespace> string = ( '"' ( [^\\"] | '\\' [wldsrnt\\"] | unicodechar )+ '"' );
<subtoken, ignore case> unicodechar = '\\x' [\dabcdef][\dabcdef][\dabcdef][\dabcdef];

<token> identifier = [\w]+;
