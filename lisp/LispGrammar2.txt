<token> number = [\d]+;

<token, mind whitespace> string = ( '"' ( [^"\\] | '\\' ["\\dlnrstw] | unicodechar )+ '"' );

<token> identifier = [\l]+;

<subtoken, ignore case> unicodechar = '\\x' [\dabcdef] [\dabcdef] [\dabcdef] [\dabcdef];
